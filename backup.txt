    // Função para gerar links aleatórios
    function getRandomLinks() {
        const randomLinks = [];
        const shuffledLinks = [...destinos].sort(() => 0.5 - Math.random()); // Embaralha os links
        const maxLinks = 5; // Quantidade de links a serem exibidos

        for (let i = 0; i < maxLinks; i++) {
            randomLinks.push(shuffledLinks[i]);
        }

        return randomLinks;
    }

    // Função para gerar números aleatórios
    function generateRandomNumbers() {
        let numbers = '';
        for (let i = 0; i < 6; i++) {
            numbers += Math.floor(Math.random() * 10);
        }
        return numbers;
    }

    // Função para gerar letras aleatórias
    function generateRandomLetters() {
        const letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz';
        let randomString = '';
        for (let i = 0; i < 6; i++) {
            randomString += letters.charAt(Math.floor(Math.random() * letters.length));
        }
        return randomString;
    }

    // Exibe os links aleatórios na página
    function displayRandomLinks() {
        const randomLinks = getRandomLinks();
        const linkList = document.getElementById('linkList');
        linkList.innerHTML = ''; // Limpa a lista antes de adicionar novos links

        randomLinks.forEach(link => {
            const listItem = document.createElement('li');
            const linkElement = document.createElement('a');
            linkElement.href = link;
            linkElement.target = '_blank';
            linkElement.textContent = link;
            listItem.appendChild(linkElement);
            linkList.appendChild(listItem);
        });
    }

    // Exibe números e letras aleatórias na página
    document.getElementById('randomNumbers').textContent = generateRandomNumbers();
    document.getElementById('randomLetters').textContent = generateRandomLetters();

    // Chama a função para exibir links aleatórios
    displayRandomLinks();
